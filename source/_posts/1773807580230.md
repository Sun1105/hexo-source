from dataclasses import dataclass, field
from abc import ABC, abstractmethod
from typing import List
import re


class Item(ABC):
    @abstractmethod
    def print(self) -> None:
        pass


@dataclass
class DataElement(Item):
    name: str
    value: str

    def print(self) -> None:
        print(f"名前：{self.name}")
        print(f"値：{self.value}")

    def getElement(self):
        return self.name + ":" + self.value


@dataclass
class Data(Item):
    name: str
    level: int
    components: List[Item] = field(default_factory=list)

    def add(self, component: Item) -> None:
        self.components.append(component)

    def remove(self, component: Item) -> None:
        self.components.remove(component)

    def get(self, name: str):
        for component in reversed(self.components):
            if component.name == name:
                return component
        return None

    def getbylevel(self, level: int):
        return_co = None
        if level == 1:
            return self
        for component in self.components:
            if hasattr(component, "components"):
                return_co = component.getbylevel(level)
            if not hasattr(component, "level"):
                continue
            if component.level == level:
                return_co = component
        return return_co

    def print(self) -> None:
        print(f"名前：{self.name}")
        for component in self.components:
            component.print()

    def getElement(self, level: int, hyphen: str):
        lines = ""
        title = self.name
        if (
            title != ""
            and len(self.components) == 1
            and hasattr(self.components[0], "name")
            and hasattr(self.components[0], "value")
            and getattr(self.components[0], "name") == title
            and str(getattr(self.components[0], "value")).lower() == "null"
        ):
            return title + ":null\n"
        if title != "":
            title = title + ":\n"

        patten = r"\((.*?)\)"
        index = ""
        if title is not None and ("[" in title or "]" in title):
            index = re.findall(patten, title)[0]
            if index != "":
                title = re.sub(patten, "", title)
            else:
                title = "\n"
                hyphen = "-"

        lines += title

        child_hyphen = hyphen
        if self.name is not None and "(リスト)" in str(self.name):
            child_hyphen = "-"

        first_in_list_item = True
        is_list_item = self.name == "" and child_hyphen == "-"

        for component in self.components:
            if component is None:
                continue
            if (not hasattr(component, "components")) or len(component.components) == 0:
                v = getattr(component, "value", None)
                if v is None or v == "" or str(v).lower() == "none":
                    continue

            if hasattr(component, "components"):
                s = component.getElement(component.level, child_hyphen)
            else:
                s = component.name + ":" + component.value + "\n"

            if s is not None and s.strip() != "":
                if (
                    hasattr(component, "components")
                    and getattr(component, "name", None) == ""
                    and child_hyphen == "-"
                ):
                    space = ""
                    prefix = ""
                    lines += space + prefix + s
                    continue

                if hasattr(component, "level"):
                    space = self.getSpace(component.level)
                else:
                    if is_list_item:
                        if first_in_list_item:
                            space = self.getSpace(level)
                        else:
                            space = self.getSpace(level + 1)
                    else:
                        space = self.getSpace(level + 1)

                prefix = child_hyphen
                if is_list_item:
                    if first_in_list_item:
                        prefix = "-"
                    else:
                        prefix = ""
                    first_in_list_item = False
                lines += space + prefix + s

        if lines.strip() == f"{self.name}:" or lines.strip() == f"{self.name}:\n":
            return ""

        return lines

    def getSpace(self, level: int):
        if level == 1:
            return "   "
        if level == 2:
            return "     "
        if level == 3:
            return "       "
        if level == 4:
            return "        "
        if level == 5:
            return "         "
        return ""
