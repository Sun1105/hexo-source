# 1. 切换到 develop 分支
git checkout develop
 
# 2. 创建新分支
git checkout -b feature/AABBBBC0001_新規作成
 
# 3. 推送新分支
git push -u origin feature/AABBBBC0001_新規作成
 
# 4. 追踪新分支(不需要，第三步完成了)
git push --set-upstream origin feature/AABBBBC0001_新規作成
 
# 5. 修改代码文件
 
# 6. 提交修改文件
git add .
git commit -m "【API機能】AABBBBC0001_新規作成"
 
# 7. 拉取远程更新
git pull
 
# 8. 将提交的修改文件，更新到远程
git push

# 9. 重命名当前分支为 
git branch -m feature/AABBBBC0001_新規作成

# 10. 删除远程旧分支（可选，确认不再使用旧分支时执行）
git push origin --delete feature/AABBBBC0001_新規作成
