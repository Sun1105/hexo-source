function bindNodeClicks() {
  const svg = document.querySelector("#cmsDiagram svg");
  if (!svg) return;

  svg.querySelectorAll(".node").forEach(node => {
    node.style.cursor = "pointer";

    let nodeName = "";
    const titleEl = node.querySelector("title");
    const labelEl = node.querySelector(".label");

    if (titleEl) {
      nodeName = titleEl.textContent.trim();
    } else if (labelEl) {
      nodeName = labelEl.textContent.trim();
    }

    if (!nodeName) return;

    node.addEventListener("click", () => {
      const desc = nodeInfo[nodeName] || "无详细信息";
      document.getElementById("description").innerHTML =
        `<h2>${nodeName}</h2><pre>${desc}</pre>`;
      document.getElementById("buttonGroup").style.display = "block";
      currentContent = desc;
    });
  });
}

// 复制按钮逻辑
let currentContent = "";
document.getElementById("copyBtn").addEventListener("click", () => {
  if (!currentContent) return alert("请先点击一个节点查看内容");
  navigator.clipboard.writeText(currentContent).then(() => {
    alert("示例内容已复制到剪贴板！");
  });
});

// 清除显示按钮
document.getElementById("clearBtn").addEventListener("click", () => {
  document.getElementById("description").innerHTML =
    `<h2>点击节点查看详细功能或API示例</h2><p>示例：点击“登录认证API”查看登录接口示例。</p>`;
  document.getElementById("buttonGroup").style.display = "none";
  currentContent = "";
});

// Mermaid 渲染完成后绑定事件
mermaid.run({ querySelector: ".mermaid" }).then(() => {
  bindNodeClicks();
});
