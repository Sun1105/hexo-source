Option Explicit                                 ' 强制变量声明，避免隐式变量导致的错误
Private RESULT_FALSE_COLOR As Long              ' 用于着色 FALSE 判定的颜色（变量，运行时赋值）

' ===== ユーザー調整オプション ==========================================
Const MATCH_BY_KEY As Boolean = False           ' 是否按“键列”突合：False=按行号突合（默认），True=按键值突合
Const KEY_COL_NAME As String = ""               ' 键列标题名（仅在 MATCH_BY_KEY=True 时生效；空则用首字段列作为键）
Const DATETIME_TOL_SEC As Double = 1            ' 日期时间比较的容差（单位：秒）
' ======================================================================

Public Sub RunBat()                             ' 主过程：抽取两表块、输出到Sheet3并比较生成判定
    Dim ws1 As Worksheet, ws2 As Worksheet, ws3 As Worksheet   ' 工作表对象引用
    Dim lastRow1 As Long, lastRow2 As Long, outRow As Long     ' 两表的最后行号、Sheet3当前输出行
    Dim tableDict1 As Object, tableDict2 As Object, tblKey As Variant  ' 存放块信息的字典、遍历键变量
    Dim tblHeaders1 As Object, tblHeaders2 As Object           ' 存放各块的表头（Variant 数组）
    Dim i As Long, key As String, sqlText As String, headerRow As Long  ' 临时变量：循环计数、键、SQL、表头行
    Dim tableName As String, colCount As Long, dataStart1 As Long, dataEnd1 As Long ' 表名、列数、块数据起止
    Dim cyanColor As Long, grayColor As Long, yellowColor As Long       ' 配色变量

    cyanColor = RGB(135, 231, 173)                       ' 青绿色（用于表头行底色）
    grayColor = RGB(221, 235, 247)                       ' 灰蓝色（用于每个块的标题行）
    yellowColor = RGB(255, 255, 0)                       ' 黄色（示例：导航行或你也可用于 FALSE）
                                                         ' 如需 FALSE 为红色，可把 RESULT_FALSE_COLOR 赋值为 RGB(255,0,0)
    RESULT_FALSE_COLOR = yellowColor                      ' 设置“FALSE 着色”为黄色；若要红色改为 RGB(255,0,0)

    Const firstColOffset As Long = 2                      ' 输出起始列偏移：结果从列B（=2）开始写

    Set ws1 = ThisWorkbook.Sheets("1")                    ' 绑定 Sheet "1"
    Set ws2 = ThisWorkbook.Sheets("2")                    ' 绑定 Sheet "2"
    Set ws3 = ThisWorkbook.Sheets("3")                    ' 绑定 Sheet "3"
    ws3.Cells.Clear                                       ' 清空 Sheet3（含数据、格式、颜色）

    Set tableDict1 = CreateObject("Scripting.Dictionary") ' 创建字典：保存Sheet1每个SQL块的元信息
    Set tblHeaders1 = CreateObject("Scripting.Dictionary")' 创建字典：保存Sheet1各块的表头行数组
    Set tableDict2 = CreateObject("Scripting.Dictionary") ' 创建字典：保存Sheet2每个SQL块的元信息
    Set tblHeaders2 = CreateObject("Scripting.Dictionary")' 创建字典：保存Sheet2各块的表头行数组

    lastRow1 = ws1.Cells(ws1.Rows.Count, 1).End(xlUp).Row  ' 求 Sheet1 的“最后有数据的行号”（按列A上探）
    lastRow2 = ws2.Cells(ws2.Rows.Count, 1).End(xlUp).Row  ' 求 Sheet2 的“最后有数据的行号”（按列A上探）
    outRow = 1                                             ' Sheet3 输出行初始为第 1 行

    ' ===== Sheet1 ブロック抽出 =================================================
    For i = 1 To lastRow1                                  ' 遍历 Sheet1 的每一行
        If Left$(CStr(ws1.Cells(i, 1).Value), 7) = "SELECT " Then  ' 若该行以 "SELECT " 开头，则视为一个块的开始
            sqlText = CStr(ws1.Cells(i, 1).Value)          ' 记录当前块的 SQL 语句（作为键）
            tableName = ""                                 ' 初始化块的表名

            ' FROM～WHERE の簡易抽出
            Dim sqlParts() As String                       ' 用于临时切分 SQL 的数组
            If InStr(1, sqlText, "FROM", vbTextCompare) > 0 Then  ' 若包含 "FROM"
                sqlParts = Split(sqlText, "FROM")          ' 以 "FROM" 切分
                If UBound(sqlParts) >= 1 Then              ' 确保存在 FROM 右侧部分
                    If InStr(1, sqlParts(1), "WHERE", vbTextCompare) > 0 Then ' 若右侧含 WHERE
                        tableName = Trim$(Split(sqlParts(1), "WHERE")(0))     ' 取 FROM 与 WHERE 间文本作为表名
                    Else
                        tableName = Trim$(sqlParts(1))     ' 否则取 FROM 右侧全部作为表名（粗略）
                    End If
                End If
            End If

            headerRow = i + 1                              ' 表头行：SQL 下方一行
            colCount = ws1.Cells(headerRow, ws1.Columns.Count).End(xlToLeft).Column ' 统计表头的最右非空列

            dataStart1 = i + 2                             ' 数据区起始：SQL 下两行
            dataEnd1 = dataStart1                          ' 数据区终止先设为起始

            ' ——以下循环：按“整行是否有值且未遇到下一个SELECT”来延伸数据区终止行——
            Do While (dataEnd1 + 1 <= lastRow1) _          ' 续行1：未超出Sheet1最后行
                     And (Left$(CStr(ws1.Cells(dataEnd1 + 1, 1).Value), 7) <> "SELECT ") _ ' 续行2：下一行不是新块的SQL
                     And (WorksheetFunction.CountA(ws1.Range(ws1.Cells(dataEnd1 + 1, 1), ws1.Cells(dataEnd1 + 1, colCount))) > 0)  ' 续行3：该行1~表头末列至少有一个值
                dataEnd1 = dataEnd1 + 1                    ' 若符合条件则把数据终止行下移一行
            Loop                                           ' 直到不满足条件为止

            key = sqlText                                  ' 以 SQL 本身作为块的唯一键
            If Not tableDict1.Exists(key) Then             ' 若字典中还没有该键
                tableDict1.Add key, Array(tableName, sqlText, headerRow, dataStart1, dataEnd1, colCount) ' 存入元信息
                tblHeaders1(key) = ws1.Cells(headerRow, 1).Resize(1, colCount).Value                     ' 取表头行的整行值（1×N 数组）
            End If
        End If
    Next i                                                 ' 下一行

    ' ===== Sheet2 ブロック抽出 =================================================
    For i = 1 To lastRow2                                  ' 遍历 Sheet2 的每一行
        If Left$(CStr(ws2.Cells(i, 1).Value), 7) = "SELECT " Then  ' 识别块开始
            sqlText = CStr(ws2.Cells(i, 1).Value)          ' 记录 SQL
            headerRow = i + 1                              ' 表头行为 SQL 下方一行
            colCount = ws2.Cells(headerRow, ws2.Columns.Count).End(xlToLeft).Column ' 表头最后列

            dataStart1 = i + 2                             ' 数据起始
            dataEnd1 = dataStart1                          ' 数据终止初始化

            ' ——同理：用“整行是否有值” + “未遇到新SELECT”来判定块的结束——
            Do While (dataEnd1 + 1 <= lastRow2) _          ' 续行1：未超出Sheet2最后行
                     And (Left$(CStr(ws2.Cells(dataEnd1 + 1, 1).Value), 7) <> "SELECT ") _ ' 续行2：下一行不是新块
                     And (WorksheetFunction.CountA(ws2.Range(ws2.Cells(dataEnd1 + 1, 1), ws2.Cells(dataEnd1 + 1, colCount))) > 0)  ' 续行3：该行1~末列至少有值
                dataEnd1 = dataEnd1 + 1                    ' 延伸数据终止行
            Loop

            key = sqlText                                  ' 以 SQL 作为块键
            If Not tableDict2.Exists(key) Then             ' 若字典中无该键
                tableDict2.Add key, Array(sqlText, headerRow, dataStart1, dataEnd1, colCount) ' 存元信息（Sheet2）
                tblHeaders2(key) = ws2.Cells(headerRow, 1).Resize(1, colCount).Value         ' 存表头行数组（Sheet2）
            End If
        End If
    Next i

    ' ===== 出力＆比較 ==========================================================
    Dim tblKeyVar As Variant                              ' 遍历块键的变量
    For Each tblKeyVar In tableDict1.Keys                 ' 遍历所有 Sheet1 的块键
        Dim arr1, arr2, header1, header2                  ' 临时：存放块信息数组、表头数组
        Dim startFieldCol1 As Long, startFieldCol2 As Long ' 每块的实际字段起始列（表头首列可能为空）
        Dim actualFieldCount As Long, actualFieldCount2 As Long ' 实际字段数（总列数-起始偏移+1）
        Dim iCol As Long                                   ' 列循环计数

        arr1 = tableDict1(tblKeyVar)                       ' 取 Sheet1 当前块元信息
        tableName = arr1(0)                                ' 表名（粗略提取）
        sqlText = arr1(1)                                  ' SQL 文本
        Dim headerRow1 As Long: headerRow1 = arr1(2)       ' 表头行（Sheet1）
        Dim dataStartA As Long: dataStartA = arr1(3)       ' 数据起始行（Sheet1）
        Dim dataEndA As Long:   dataEndA = arr1(4)         ' 数据终止行（Sheet1）
        Dim colCount1 As Long:  colCount1 = arr1(5)        ' 列数（Sheet1）
        header1 = tblHeaders1(tblKeyVar)                   ' 表头行数组（Sheet1）

        Dim hasTable2 As Boolean: hasTable2 = tableDict2.Exists(tblKeyVar) ' Sheet2 是否存在对应块

        ' 見出しラベル
        ws3.Cells(outRow, 1).Value = tableName             ' 输出块标题：表名
        ws3.Rows(outRow).Interior.Color = grayColor        ' 标题行底色（灰蓝）
        outRow = outRow + 1                                ' 下一行
        ws3.Cells(outRow, 1).Value = sqlText               ' 输出 SQL 文本
        outRow = outRow + 1                                ' 下一行

        ' ナビ行（如需保留，可取消注释）
        ' ws3.Cells(outRow, 1).Value = "Ctrl+→"           ' 导航提示（可选）
        ' ws3.Cells(outRow, firstColOffset).Value = "★"    ' 导航标记（可选）
        ' ws3.Range(ws3.Cells(outRow, 1), ws3.Cells(outRow, colCount1 + 1)).Interior.Color = yellowColor ' 导航行底色
        ' outRow = outRow + 1                              ' 导航行结束，移到下一行

        ' ==== Sheet1 のヘッダー ====
        startFieldCol1 = 1                                 ' 假设字段从表头第1列开始
        If Trim$(CStr(header1(1, 1) & "")) = "" Then startFieldCol1 = 2  ' 若表头第1列为空，则从第2列开始

        actualFieldCount = colCount1 - startFieldCol1 + 1  ' 实际字段数（去掉可能的前导空列）
        For iCol = 0 To actualFieldCount - 1               ' 遍历所有字段列
            With ws3.Cells(outRow, iCol + firstColOffset)  ' 定位到 Sheet3 的输出单元格
                .Value = ws1.Cells(headerRow1, startFieldCol1 + iCol).Value  ' 写入表头文字
                .NumberFormat = "@"                        ' 表头强制为文本
            End With
        Next iCol
        ws3.Range(ws3.Cells(outRow, firstColOffset), ws3.Cells(outRow, firstColOffset + actualFieldCount - 1)).Interior.Color = cyanColor ' 表头行底色（青绿）
        Dim headerVal1 As Long: headerVal1 = outRow        ' 记录 Sheet1 表头所在的行号
        outRow = outRow + 1                                 ' 移到数据区第一行

        ' ==== Sheet1 のデータ ====
        Dim dataCount1 As Long                             ' Sheet1 数据行数
        dataCount1 = dataEndA - dataStartA + 1             ' 计算数据行数

        Dim dRow As Long, lastDataRow1 As Long             ' 数据区循环行号、最后一行记录
        For dRow = 0 To dataCount1 - 1                     ' 遍历每一条记录
            For iCol = 0 To actualFieldCount - 1           ' 遍历每一个字段
                Dim src1 As Range, dst1 As Range           ' 源单元格、目标单元格
                Set src1 = ws1.Cells(dataStartA + dRow, startFieldCol1 + iCol) ' 源：Sheet1 对应单元格
                Set dst1 = ws3.Cells(outRow, iCol + firstColOffset)            ' 目标：Sheet3 输出位置
                dst1.Value = src1.Value                    ' 复制“值”（保持类型）
                On Error Resume Next                       ' 尝试复制显示格式，出错则忽略
                dst1.NumberFormatLocal = src1.NumberFormatLocal  ' 同步源的本地化格式（日期等保持显示）
                On Error GoTo 0                            ' 恢复错误处理
            Next iCol
            outRow = outRow + 1                            ' 填完一行后下移
        Next dRow
        lastDataRow1 = outRow - 1                          ' 记录 Sheet1 数据块在 Sheet3 的最后行

        ' 罫線（为表头+数据区画边框）
        With ws3.Range(ws3.Cells(headerVal1, firstColOffset), ws3.Cells(lastDataRow1, firstColOffset + actualFieldCount - 1)).Borders
            .LineStyle = xlContinuous                      ' 连续线
            .Weight = xlThin                               ' 细线
        End With

        outRow = outRow + 1                                ' Sheet1 块与下方内容之间空一行

        ' ==== Sheet2 がある場合 ====
        Dim headerRow2 As Long, dataStartB As Long, dataEndB As Long, colCount2 As Long ' Sheet2 对应块的元信息
        If hasTable2 Then                                  ' 若 Sheet2 存在对应块
            arr2 = tableDict2(tblKeyVar)                   ' 取该块元信息数组
            headerRow2 = arr2(1)                           ' 表头行（Sheet2）
            dataStartB = arr2(2)                           ' 数据起始行（Sheet2）
            dataEndB = arr2(3)                             ' 数据终止行（Sheet2）
            colCount2 = arr2(4)                            ' 列数（Sheet2）
            header2 = tblHeaders2(tblKeyVar)               ' 表头数组（Sheet2）

            startFieldCol2 = 1                             ' 默认从第1列表头开始
            If Trim$(CStr(header2(1, 1) & "")) = "" Then startFieldCol2 = 2  ' 若第1列表头为空，从第2列起
            actualFieldCount2 = colCount2 - startFieldCol2 + 1               ' 实际字段数（Sheet2）

            ws3.Cells(outRow, 1).Value = sqlText           ' 再输出一次 SQL（用于区分上下两表）
            outRow = outRow + 1                            ' 下一行

            ' Sheet2 ヘッダー
            For iCol = 0 To actualFieldCount2 - 1          ' 遍历字段列
                With ws3.Cells(outRow, iCol + firstColOffset)    ' 定位输出单元格
                    .Value = ws2.Cells(headerRow2, startFieldCol2 + iCol).Value ' 写入表头文字
                    .NumberFormat = "@"                    ' 表头设为文本
                End With
            Next iCol
            ws3.Range(ws3.Cells(outRow, firstColOffset), ws3.Cells(outRow, firstColOffset + actualFieldCount2 - 1)).Interior.Color = cyanColor ' 表头底色
            Dim headerVal2 As Long: headerVal2 = outRow    ' 记录 Sheet2 表头所在行
            outRow = outRow + 1                            ' 移到数据区起始行

            ' Sheet2 データ
            Dim dataCount2 As Long                         ' Sheet2 数据行数
            dataCount2 = dataEndB - dataStartB + 1         ' 计算数据行数
            Dim curDataRow2Start As Long: curDataRow2Start = outRow  ' 记录数据区开始行（备用）

            For dRow = 0 To dataCount2 - 1                 ' 遍历每一条记录
                For iCol = 0 To actualFieldCount2 - 1      ' 遍历每一列
                    Dim src2 As Range, dst2 As Range       ' 源/目标单元格
                    Set src2 = ws2.Cells(dataStartB + dRow, startFieldCol2 + iCol) ' 源：Sheet2
                    Set dst2 = ws3.Cells(outRow, iCol + firstColOffset)            ' 目标：Sheet3
                    dst2.Value = src2.Value                ' 复制值（保持类型）
                    On Error Resume Next                   ' 尝试复制显示格式
                    dst2.NumberFormatLocal = src2.NumberFormatLocal ' 同步源格式
                    On Error GoTo 0                        ' 恢复错误处理
                Next iCol
                outRow = outRow + 1                        ' 数据行下移
            Next dRow

            Dim lastDataRow2 As Long: lastDataRow2 = outRow - 1 ' 记录 Sheet2 数据块在 Sheet3 的最后行
            With ws3.Range(ws3.Cells(headerVal2, firstColOffset), ws3.Cells(lastDataRow2, firstColOffset + actualFieldCount2 - 1)).Borders
                .LineStyle = xlContinuous                  ' 连续线
                .Weight = xlThin                           ' 细线
            End With

            ' ===== 判定 =====
            Dim maxRow As Long                             ' 计算两表数据行数的最大值
            maxRow = WorksheetFunction.Max(dataCount1, dataCount2) ' 取较大者，逐行对齐比较
            outRow = outRow + 1                            ' 判定区上方空一行

            If MATCH_BY_KEY Then                           ' 如果启用“按键值突合”
                ' ---------- キー突合せモード ----------
                Dim keyColIndex1 As Long, keyColIndex2 As Long                       ' 键列的绝对列号（两表）
                keyColIndex1 = ResolveKeyColumnIndex(ws1, headerRow1, startFieldCol1, actualFieldCount, KEY_COL_NAME) ' 解析Sheet1键列
                keyColIndex2 = ResolveKeyColumnIndex(ws2, headerRow2, startFieldCol2, actualFieldCount2, KEY_COL_NAME) ' 解析Sheet2键列

                Dim map1 As Object, map2 As Object                                   ' 键 → 行号 的映射字典
                Set map1 = BuildKeyToRowMap(ws1, dataStartA, dataEndA, keyColIndex1) ' 由Sheet1构建映射（重复键取最后）
                Set map2 = BuildKeyToRowMap(ws2, dataStartB, dataEndB, keyColIndex2) ' 由Sheet2构建映射

                Dim uni As Object: Set uni = UnionKeys(map1, map2)                   ' 两表键的并集

                Dim k As Variant, r1 As Long, r2 As Long                             ' 遍历每个键并取对应行
                For Each k In uni.Keys
                    If ws3.Cells(outRow, 1).Value = "" Then ws3.Cells(outRow, 1).Value = "判定 (Key=" & k & ")" ' 首行写“判定”标题（含键）
                    r1 = NzLng(map1(k))                                              ' Sheet1 该键所在行（0 表示无）
                    r2 = NzLng(map2(k))                                              ' Sheet2 该键所在行（0 表示无）
                    WriteCompareRow ws3, ws1, ws2, outRow, firstColOffset, _         ' 写出一行判定（逐列 TRUE/FALSE）
                                    headerRow1, startFieldCol1, actualFieldCount, r1, _
                                    headerRow2, startFieldCol2, actualFieldCount2, r2
                    outRow = outRow + 1                                              ' 判定行下移
                Next k

            Else                                           ' 否则按行号一一对应比较
                For dRow = 0 To maxRow - 1                 ' 从第 0 条到 maxRow-1 条记录
                    If dRow = 0 Then ws3.Cells(outRow, 1).Value = "判定" ' 首行在A列输出“判定”
                    WriteCompareRow ws3, ws1, ws2, outRow, firstColOffset, _         ' 写出一行判定（逐列）
                                    headerRow1, startFieldCol1, actualFieldCount, dataStartA + dRow, _
                                    headerRow2, startFieldCol2, actualFieldCount2, dataStartB + dRow
                    outRow = outRow + 1                                              ' 判定行下移
                Next dRow
            End If

            outRow = outRow + 1                              ' 判定区与下一个块之间空一行

        Else                                                 ' 若 Sheet2 未找到对应块
            ws3.Cells(outRow, 1).Value = "Sheet2見つかりません" ' 在A列提示“Sheet2未找到”
            outRow = outRow + 2                               ' 空两行
        End If

        outRow = outRow + 1                                    ' 每个块之间再空一行
    Next tblKeyVar                                            ' 遍历下一个块键

    ws3.Cells.Font.Name = "Meiryo UI"                          ' 设置 Sheet3 使用 Meiryo UI 字体

    MsgBox "Sheet3導出完了"                                     ' 弹窗提示：导出完成
End Sub

Private Sub WriteCompareRow(ByVal wsOut As Worksheet, _        ' 子过程：写出一行“判定”结果（逐列 TRUE/FALSE）
                            ByVal ws1 As Worksheet, ByVal ws2 As Worksheet, _
                            ByVal outRow As Long, ByVal firstColOffset As Long, _
                            ByVal headerRow1 As Long, ByVal startFieldCol1 As Long, ByVal fieldCount1 As Long, ByVal row1 As Long, _
                            ByVal headerRow2 As Long, ByVal startFieldCol2 As Long, ByVal fieldCount2 As Long, ByVal row2 As Long)

    Dim iCol As Long                                            ' 列循环计数
    For iCol = 0 To fieldCount1 - 1                             ' 遍历 Sheet1 的每一个字段列（作为基准列数）
        Dim v1 As Variant, v2 As Variant                        ' 两侧待比较的值

        If row1 >= headerRow1 + 1 Then v1 = ws1.Cells(row1, startFieldCol1 + iCol).Value Else v1 = Empty ' 若行存在则取 Sheet1 值，否则 Empty
        If row2 >= headerRow2 + 1 Then                           ' 若 Sheet2 该行存在
            If iCol <= fieldCount2 - 1 Then                      ' 且该列在 Sheet2 的字段范围内
                v2 = ws2.Cells(row2, startFieldCol2 + iCol).Value ' 取 Sheet2 值
            Else
                v2 = Empty                                        ' 否则视为空
            End If
        Else
            v2 = Empty                                            ' 若 Sheet2 行不存在则为空
        End If

        Dim isEq As Boolean                                      ' 该列是否相等
        isEq = AreEqualValue(v1, v2)                              ' 调用比较函数判断（日期容差、0与空、数值/字符串等规则）

        With wsOut.Cells(outRow, iCol + firstColOffset)           ' 定位到判定单元格
            .Value = IIf(isEq, "TRUE", "FALSE")                   ' 写入字符串 "TRUE"/"FALSE"
            .NumberFormat = "@"                                   ' 判定结果作为文本处理
            If Not isEq Then
                .Interior.Color = RESULT_FALSE_COLOR              ' 若不相等则按预设颜色着色（如黄色或红色）
            Else
                .Interior.Pattern = xlNone                        ' 相等则清除填充（可按需改成绿色等）
            End If
        End With
    Next iCol                                                     ' 下一列
End Sub

Private Function AreEqualValue(ByVal v1 As Variant, ByVal v2 As Variant) As Boolean  ' 值比较函数（含多种规则）
    ' --- 规则0：0 与 空必定不相等 ---
    If (IsZeroLike(v1) And IsEmptyLike(v2)) Or (IsZeroLike(v2) And IsEmptyLike(v1)) Then
        AreEqualValue = False                                     ' 一边为0，一边为空 => 直接 False
        Exit Function
    End If

    ' --- 规则1：双方都为空 => 相等 ---
    If IsEmptyLike(v1) And IsEmptyLike(v2) Then
        AreEqualValue = True                                      ' 两边都空 => True
        Exit Function
    End If

    ' --- 规则2：日期/时间比较（按序列值，支持秒级容差） ---
    If IsDateLike(v1) Or IsDateLike(v2) Then
        On Error GoTo Fallback                                    ' 尝试转为 Double（序列值），失败转入后备
        Dim d1 As Double, d2 As Double
        d1 = CDbl(v1): d2 = CDbl(v2)                              ' 转为序列值
        AreEqualValue = (Abs(d1 - d2) <= (DATETIME_TOL_SEC / 86400#)) ' 容差：秒÷一天（86400）
        Exit Function
Fallback:
        ' 若无法转为数值（例如字符串日期格式异常），则落回到后面的字符串比较
    End If

    ' --- 规则3：双方皆为数值 => 数值比较 ---
    If IsNumeric(v1) And IsNumeric(v2) Then
        AreEqualValue = (CDbl(v1) = CDbl(v2))                     ' 转为双精度再比较
        Exit Function
    End If

    ' --- 规则4：字符串比较（去空格、"NULL" 视为空、大小写不敏感） ---
    Dim s1 As String, s2 As String
    s1 = NormalizeString(v1)                                      ' 归一化字符串1
    s2 = NormalizeString(v2)                                      ' 归一化字符串2
    AreEqualValue = (StrComp(s1, s2, vbTextCompare) = 0)          ' 忽略大小写比较
End Function

Private Function NormalizeString(ByVal v As Variant) As String    ' 字符串归一化（空/NULL→空串）
    If IsEmptyLike(v) Then
        NormalizeString = ""                                      ' 空值统一为空串
    Else
        NormalizeString = Trim$(CStr(v))                          ' 转字符串并修剪空格
        If UCase$(NormalizeString) = "NULL" Then NormalizeString = ""  ' 字面 "NULL" 也当空
    End If
End Function

Private Function IsEmptyLike(ByVal v As Variant) As Boolean       ' 判断“空值”集合
    IsEmptyLike = (IsEmpty(v) Or IsNull(v) Or (CStr(v) = "") Or (UCase$(Trim$(CStr(v))) = "NULL")) ' 空/Null/空串/"NULL"
End Function

Private Function IsDateLike(ByVal v As Variant) As Boolean        ' 宽松判断是否可当作日期
    On Error Resume Next                                          ' 先尝试转为数值（Excel 日期序列）
    Dim tmp As Double: tmp = CDbl(v)
    IsDateLike = (Err.Number = 0 And tmp > 0)                     ' 能转且>0 视为可能是日期序列
    Err.Clear
    On Error GoTo 0
    If Not IsDateLike Then IsDateLike = IsDate(v)                 ' 否则再用 VBA 的 IsDate 兜底判断
End Function

Private Function ResolveKeyColumnIndex(ByVal ws As Worksheet, ByVal headerRow As Long, ByVal startFieldCol As Long, ByVal fieldCount As Long, ByVal keyHeader As String) As Long
                                                                 ' 解析“键列”的绝对列号：优先匹配标题名，否则回退到首字段列
    Dim idx As Long
    If LenB(keyHeader) = 0 Then
        ResolveKeyColumnIndex = startFieldCol                     ' 未给标题名则直接用首字段列
        Exit Function
    End If
    For idx = 0 To fieldCount - 1                                 ' 遍历字段列，寻找与标题名一致的列
        If StrComp(CStr(ws.Cells(headerRow, startFieldCol + idx).Value), keyHeader, vbTextCompare) = 0 Then
            ResolveKeyColumnIndex = startFieldCol + idx           ' 找到匹配列，返回绝对列号
            Exit Function
        End If
    Next idx
    ResolveKeyColumnIndex = startFieldCol                         ' 未找到则回退到首字段列
End Function

Private Function BuildKeyToRowMap(ByVal ws As Worksheet, ByVal dataStart As Long, ByVal dataEnd As Long, ByVal keyCol As Long) As Object
                                                                 ' 构建“键 → 行号”的映射（若有重复键，采用最后一次出现的行）
    Dim d As Object: Set d = CreateObject("Scripting.Dictionary")
    Dim r As Long, k As Variant
    For r = dataStart To dataEnd                                  ' 遍历数据区每一行
        k = ws.Cells(r, keyCol).Value                             ' 取该行键值
        If Not IsEmptyLike(k) Then d(CStr(k)) = r                 ' 非空则记录（字符串化键）
    Next r
    Set BuildKeyToRowMap = d                                      ' 返回字典
End Function

Private Function UnionKeys(ByVal d1 As Object, ByVal d2 As Object) As Object
                                                                 ' 取两个键字典的并集（只关心键名）
    Dim u As Object: Set u = CreateObject("Scripting.Dictionary")
    Dim k As Variant
    For Each k In d1.Keys: u(k) = True: Next k                   ' 先放入字典1的键
    For Each k In d2.Keys: u(k) = True: Next k                   ' 再放入字典2的键（重复自动覆盖为同一键）
    Set UnionKeys = u                                             ' 返回并集字典
End Function

Private Function NzLng(ByVal v As Variant) As Long               ' 将可空变体转为 Long（Empty→0，否则转 Long）
    If IsEmpty(v) Then NzLng = 0 Else NzLng = CLng(v)
End Function

Private Function IsZeroLike(ByVal v As Variant) As Boolean       ' 判断是否为“0”（数值0或字符串"0"）
    If IsEmpty(v) Or IsNull(v) Then
        IsZeroLike = False                                       ' 空/Null 不是 0
    ElseIf IsNumeric(v) Then
        IsZeroLike = (CDbl(v) = 0)                               ' 数值型且等于 0
    Else
        IsZeroLike = (Trim$(CStr(v)) = "0")                      ' 字符串内容正好是 "0"
    End If
End Function

                sqlParts = Split(sqlText, "FROM")

                    If InStr(1, sqlParts(1), "WHERE", vbTextCompare) > 0 Then

            dataEnd1 = dataStart1

                     And (WorksheetFunction.CountA(ws2.Range(ws2.Cells(dataEnd1 + 1, 1), ws2.Cells(dataEnd1 + 1, colCount))) > 0)
                dataEnd1 = dataEnd1 + 1
            Loop

Private Sub WriteCompareRow(ByVal wsOut As Worksheet, _
                            ByVal ws1 As Worksheet, ByVal ws2 As Worksheet, _
                            ByVal outRow As Long, ByVal firstColOffset As Long, _
                            ByVal headerRow1 As Long, ByVal startFieldCol1 As Long, ByVal fieldCount1 As Long, ByVal row1 As Long, _
                            ByVal headerRow2 As Long, ByVal startFieldCol2 As Long, ByVal fieldCount2 As Long, ByVal row2 As Long)
    Dim iCol As Long
    For iCol = 0 To fieldCount1 - 1
        Dim v1 As Variant, v2 As Variant

        If row1 >= headerRow1 + 1 Then v1 = ws1.Cells(row1, startFieldCol1 + iCol).Value Else v1 = Empty
        If row2 >= headerRow2 + 1 Then
            If iCol <= fieldCount2 - 1 Then
                v2 = ws2.Cells(row2, startFieldCol2 + iCol).Value
            Else
                v2 = Empty
            End If
        Else
            v2 = Empty
        End If

        wsOut.Cells(outRow, iCol + firstColOffset).Value = IIf(AreEqualValue(v1, v2), "TRUE", "FALSE")
        wsOut.Cells(outRow, iCol + firstColOffset).NumberFormat = "@"
    Next iCol
End Sub

Private Function AreEqualValue(ByVal v1 As Variant, ByVal v2 As Variant) As Boolean
    If IsEmptyLike(v1) And IsEmptyLike(v2) Then
        AreEqualValue = True
        Exit Function
    End If

    If IsDateLike(v1) Or IsDateLike(v2) Then
        On Error GoTo Fallback
        Dim d1 As Double, d2 As Double
        d1 = CDbl(v1): d2 = CDbl(v2)
        AreEqualValue = (Abs(d1 - d2) <= (DATETIME_TOL_SEC / 86400#))
        Exit Function
Fallback:
    End If

    If IsNumeric(v1) And IsNumeric(v2) Then
        AreEqualValue = (CDbl(v1) = CDbl(v2))
        Exit Function
    End If

    Dim s1 As String, s2 As String
    s1 = NormalizeString(v1)
    s2 = NormalizeString(v2)
    AreEqualValue = (StrComp(s1, s2, vbTextCompare) = 0)
End Function

Private Function NormalizeString(ByVal v As Variant) As String
    If IsEmptyLike(v) Then
        NormalizeString = ""
    Else
        NormalizeString = Trim$(CStr(v))
        If UCase$(NormalizeString) = "NULL" Then NormalizeString = ""
    End If
End Function

Private Function IsEmptyLike(ByVal v As Variant) As Boolean
    IsEmptyLike = (IsEmpty(v) Or IsNull(v) Or (CStr(v) = "") Or (UCase$(Trim$(CStr(v))) = "NULL"))
End Function

Private Function IsDateLike(ByVal v As Variant) As Boolean
    On Error Resume Next
    Dim tmp As Double: tmp = CDbl(v)
    IsDateLike = (Err.Number = 0 And tmp > 0)
    Err.Clear
    On Error GoTo 0
    If Not IsDateLike Then IsDateLike = IsDate(v)
End Function

    ResolveKeyColumnIndex = startFieldCol
