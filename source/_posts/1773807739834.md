import logging

from openpyxl import load_workbook

from CommonUtils import CommonUtils
from testcase_dto import Item, Data, DataElement
from constants import PCL_INPUT_HEADERS, PCL_OUTPUT_HEADERS


def create_yaml(input_dir_path):
    sheet_name = "テストデータ"

    workbook = load_workbook(input_dir_path)
    ws = workbook[sheet_name]

    class_name = ws["I1"].value
    method_name = ws["I2"].value

    labels = []
    if isinstance(PCL_INPUT_HEADERS, list):
        labels.extend([x for x in PCL_INPUT_HEADERS if str(x).strip() != ""])
    if isinstance(PCL_OUTPUT_HEADERS, list):
        labels.extend([x for x in PCL_OUTPUT_HEADERS if str(x).strip() != ""])

    section_rows: list[tuple[int, str]] = []
    for label in labels:
        for r in range(1, ws.max_row + 1):
            v = ws.cell(row=r, column=1).value
            if v is not None and str(v).strip() == label:
                section_rows.append((r, label))
                break
    section_rows.sort(key=lambda x: x[0])

    checkid_row = None
    checkid_col = None
    for r in range(1, ws.max_row + 1):
        found = False
        for c in range(1, ws.max_column + 1):
            v = ws.cell(row=r, column=c).value
            if v is not None and str(v).strip() == "チェックID":
                checkid_row = r
                checkid_col = c
                found = True
                break
        if found:
            break
    if checkid_row is None or checkid_col is None:
        return "", class_name, method_name

    checkids: list[tuple[str, int]] = []
    for c in range(checkid_col, ws.max_column + 1):
        v = ws.cell(row=checkid_row + 1, column=c).value
        if v is None or str(v).strip() == "":
            break
        checkids.append((str(v).strip(), c))

    root = Data("", 0)
    cases: dict[str, dict] = {}
    for cid, _ccol in checkids:
        case_node = Data(cid, 1)
        case_node.add(DataElement("チェックID", cid))
        input_node = Data("入力条件", 2)
        output_node = Data("出力内容", 2)
        case_node.add(input_node)
        case_node.add(output_node)
        root.add(case_node)
        cases[cid] = {"in": input_node, "out": output_node}

    for idx, (start_r, label) in enumerate(section_rows):
        end_r = (section_rows[idx + 1][0] - 1) if idx + 1 < len(section_rows) else ws.max_row

        local_row = None
        local_col = None
        for r in range(start_r, min(end_r, start_r + 10) + 1):
            for c in range(1, ws.max_column + 1):
                v = ws.cell(row=r, column=c).value
                if v is not None and str(v).strip() == "チェックID":
                    local_row = r
                    local_col = c
                    break
            if local_row is not None:
                break

        use_checkid_row = local_row if local_row is not None else checkid_row
        use_checkid_col = local_col if local_col is not None else checkid_col
        if use_checkid_row is None or use_checkid_col is None:
            continue

        section_ids: list[tuple[str, int]] = []
        for c in range(use_checkid_col, ws.max_column + 1):
            v = ws.cell(row=use_checkid_row + 1, column=c).value
            if v is None or str(v).strip() == "":
                break
            section_ids.append((str(v).strip(), c))
        if not section_ids:
            continue

        class_row = None
        for r in range(use_checkid_row + 2, end_r + 1):
            v = ws.cell(row=r, column=2).value
            if v is None:
                continue
            s = str(v).strip()
            if s == "" or s == "データ項目":
                continue
            class_row = r
            break
        if class_row is None:
            continue
        cls_name = str(ws.cell(row=class_row, column=2).value).strip()

        is_input = isinstance(PCL_INPUT_HEADERS, list) and label in PCL_INPUT_HEADERS
        is_output = isinstance(PCL_OUTPUT_HEADERS, list) and label in PCL_OUTPUT_HEADERS

        is_exception = ("例外" in str(label)) or str(label).startswith("4.")

        for cid, ccol in section_ids:
            if is_input:
                header_node = Data(label, 2)
                class_node = Data(cls_name, 2)
                v = ws.cell(row=class_row, column=ccol).value
                if v is not None and str(v).strip().lower() == "null":
                    class_node.add(DataElement(cls_name, "null"))
                else:
                    for r in range(class_row + 1, end_r + 1):
                        f = ws.cell(row=r, column=3).value
                        if f is None or str(f).strip() == "":
                            continue
                        fname = str(f).strip()
                        if fname == "QRコ一ド番号":
                            fname = "QRコード番号"
                        val = ws.cell(row=r, column=ccol).value
                        if val is None:
                            continue
                        sval = str(val).strip()
                        if sval == "" or sval.lower() == "null":
                            continue
                        class_node.add(DataElement(fname, sval))
                header_node.add(class_node)
                cases[cid]["in"].add(header_node)

            elif is_output and (not is_exception):
                out_header_node = Data(label, 2)
                out_class_node = Data(cls_name, 2)
                v = ws.cell(row=class_row, column=ccol).value
                if v is not None and str(v).strip().lower() == "null":
                    out_class_node.add(DataElement(cls_name, "null"))
                    out_header_node.add(out_class_node)
                    cases[cid]["out"].add(out_header_node)
                    continue

                current_list_item = None
                list_node = None

                for r in range(class_row + 1, end_r + 1):
                    title_c = ws.cell(row=r, column=3).value
                    title_d = ws.cell(row=r, column=4).value
                    if title_c is not None:
                        tc = str(title_c).strip()
                    else:
                        tc = ""

                    val = ws.cell(row=r, column=ccol).value
                    if val is None:
                        continue
                    sval = str(val).strip()
                    if sval == "" or sval.lower() == "null":
                        continue

                    if tc.endswith("(リスト)"):
                        if list_node is None:
                            list_node = Data(tc, 3)
                            out_class_node.add(list_node)

                    if list_node is not None and sval.startswith("[") and sval.endswith("]"):
                        current_list_item = Data("", 4)
                        list_node.add(current_list_item)
                        continue

                    if title_d is not None and current_list_item is not None:
                        current_list_item.add(DataElement(str(title_d).strip(), sval))
                        continue

                    if tc != "" and (not tc.endswith("(リスト)")):
                        fname = tc
                        if fname == "QRコ一ド番号":
                            fname = "QRコード番号"
                        out_class_node.add(DataElement(fname, sval))

                out_header_node.add(out_class_node)
                cases[cid]["out"].add(out_header_node)

            elif is_output and is_exception:
                ex_header_node = Data(label, 2)
                ex_node = Data(cls_name, 2)
                msg_row = None
                for r in range(class_row + 1, end_r + 1):
                    if ws.cell(row=r, column=3).value is not None:
                        msg_row = r
                        break
                if msg_row is not None:
                    msg_title = ws.cell(row=msg_row, column=3).value
                    msg_val = ws.cell(row=msg_row, column=ccol).value
                    if msg_title is not None and msg_val is not None and str(msg_val).strip() != "":
                        ex_node.add(DataElement(str(msg_title).strip(), str(msg_val).strip()))
                        ex_header_node.add(ex_node)
                        cases[cid]["out"].add(ex_header_node)

    yaml_text = root.getElement(0, "")
    CommonUtils.writeLog(
        logging.INFO,
        f"テストケース数:{len(root.components)} / PCLファイル【{input_dir_path}】",
    )

    return yaml_text, class_name, method_name


def get_title(sheet, cell, i):
    for row in sheet.iter_rows(min_row=cell.row, max_row=cell.row):
        for cell2 in row:
            if cell2.column == cell.column - i:
                return None
            if cell2.value is not None:
                return cell2
    return None


def isCheckId(sheet, cell):
    if cell.value == "チェックID":
        return True
    if cell.value is None:
        return False
    for row in sheet.iter_rows(
        min_row=cell.row, max_row=cell.row, max_col=cell.column
    ):
        for cell2 in row:
            if cell2.value == "チェックID":
                return True
    return False
