    function escapeHTML(str){
      return String(str).replace(/[&<>"']/g, s => ({
        "&":"&amp;","<":"&lt;",">":"&gt;","\"":"&quot;","'":"&#39;"
      }[s]));
    }

    function renderTabs(){
      const btns = document.getElementById("tabButtons");
      const panels = document.getElementById("tabPanels");
      btns.innerHTML = "";
      panels.innerHTML = "";

      coreDocs.forEach((group, i) => {
        const btn = document.createElement("button");
        btn.textContent = group.category;
        if(i === 0) btn.classList.add("active");
        btn.onclick = () => {
          document.querySelectorAll(".tab-buttons button").forEach(b => b.classList.remove("active"));
          document.querySelectorAll(".tab-panel").forEach(p => p.classList.remove("active"));
          btn.classList.add("active");
          document.getElementById("panel-" + i).classList.add("active");
        };
        btns.appendChild(btn);

        const panel = document.createElement("div");
        panel.className = "tab-panel" + (i === 0 ? " active" : "");
        panel.id = "panel-" + i;
        panel.innerHTML = `
          <div class="card doc">
            <h3>${group.icon} ${escapeHTML(group.category)}</h3>
            <div class="doc-list">
              ${group.items.map(([name,note]) => `
                <div class="doc-item">
                  <b>□ ${escapeHTML(name)}</b>
                  <span>${escapeHTML(note)}</span>
                </div>
              `).join("")}
            </div>
          </div>`;
        panels.appendChild(panel);
      });
    }

    function renderPlaces(rows){
      document.getElementById("placesBody").innerHTML = rows.map(([doc,unit,url,site,note]) => {
        const isHttp = /^https?:\/\//.test(url);
        const siteHtml = isHttp
          ? `<a class="site-link" href="${escapeHTML(url)}" target="_blank" rel="noopener noreferrer">${escapeHTML(site)}</a><span class="subnote">${escapeHTML(url)}</span>`
          : `<span class="unit">${escapeHTML(site)}</span><span class="subnote">${escapeHTML(url)}</span>`;
        return `<tr>
          <td>${escapeHTML(doc)}</td>
          <td><span class="unit">${escapeHTML(unit)}</span></td>
          <td>${siteHtml}</td>
          <td>${escapeHTML(note)}</td>
        </tr>`;
      }).join("");
    }

    function renderChecklist(){
      const body = document.getElementById("checkBody");
      const total = checklist.length;
      document.getElementById("totalCount").textContent = total;
      body.innerHTML = checklist.map((item, i) => `
        <label class="check">

          <span>${escapeHTML(item)}</span>
        </label>
      `).join("");

      function update(){
        const done = document.querySelectorAll("[data-check]:checked").length;
        document.getElementById("doneCount").textContent = done;
        document.getElementById("progressBar").style.width = (done / total * 100) + "%";
      }
