package com.company.infrastructure.http; // 定义企业基础设施层HTTP包

import java.io.IOException; // IO异常
import java.net.URI; // URI对象
import java.net.URLEncoder; // URL编码工具
import java.net.http.HttpClient; // JDK11 HttpClient
import java.net.http.HttpRequest; // HTTP请求
import java.net.http.HttpResponse; // HTTP响应
import java.nio.charset.StandardCharsets; // 标准字符集
import java.time.Duration; // 时间间隔类
import java.util.HashMap; // HashMap实现
import java.util.Map; // Map接口
import java.util.StringJoiner; // 字符串拼接工具
import java.util.concurrent.TimeUnit; // 时间单位
import java.util.function.Supplier; // 函数式接口

/**
 * ============================================================
 * 企业级HTTP基盘组件
 * ============================================================
 *
 * 设计原则：
 * 1. 无状态（线程安全）
 * 2. 单例HttpClient
 * 3. Builder模式
 * 4. 指数退避重试
 * 5. 统一响应模型
 */
public final class EnterpriseHttpClient { // 定义最终HTTP执行器类（final禁止继承）

    /**
     * 单例HttpClient
     * HttpClient是线程安全的，可以全局复用
     */
    private static final HttpClient CLIENT = // 定义静态常量HttpClient
            HttpClient.newBuilder() // 创建Builder
                    .version(HttpClient.Version.HTTP_2) // 启用HTTP2
                    .connectTimeout(Duration.ofSeconds(10)) // 默认连接超时10秒
                    .build(); // 构建Client

    /**
     * 私有构造函数
     * 防止外部实例化（工具类模式）
     */
    private EnterpriseHttpClient() { }

    /**
     * 对外统一执行入口
     *
     * @param requestSupplier  请求构造函数（支持每次重试重新生成）
     * @param maxRetries       最大重试次数
     * @return                 HTTP响应结果
     */
    public static HttpResult execute( // 静态执行方法
            Supplier<HttpRequest> requestSupplier, // 使用Supplier保证每次重试重新构建请求
            int maxRetries) { // 最大重试次数

        for (int attempt = 0; attempt < maxRetries; attempt++) { // 重试循环

            try {
                HttpRequest request = requestSupplier.get(); // 每次重新获取请求对象

                HttpResponse<String> response =
                        CLIENT.send(request, // 发送请求
                                HttpResponse.BodyHandlers.ofString()); // 响应转为字符串

                return new HttpResult( // 返回统一响应对象
                        response.statusCode(), // 状态码
                        response.body(), // 响应体
                        response.headers().map()); // 响应头

            } catch (IOException | InterruptedException ex) { // 捕获通信异常

                if (attempt == maxRetries - 1) { // 如果达到最大重试次数
                    throw new RuntimeException("HTTP请求失败", ex); // 抛出异常
                }

                sleepExponential(attempt); // 指数退避等待
            }
        }

        throw new IllegalStateException("不可达代码"); // 理论上不会执行到这里
    }

    /**
     * 指数退避算法
     *
     * 第1次：1秒
     * 第2次：2秒
     * 第3次：4秒
     */
    private static void sleepExponential(int attempt) { // 退避函数

        try {
            long delay = (long) Math.pow(2, attempt); // 计算2的n次方秒
            TimeUnit.SECONDS.sleep(delay); // 睡眠等待
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt(); // 恢复线程中断状态
        }
    }

    /**
     * ===============================
     * 请求构建器（Builder模式）
     * ===============================
     */
    public static class RequestBuilder { // 定义请求构建器

        private String url; // 请求URL
        private String method = "GET"; // 默认GET
        private Duration timeout = Duration.ofSeconds(10); // 默认超时
        private final Map<String, String> headers = new HashMap<>(); // 请求头
        private final Map<String, String> params = new HashMap<>(); // 参数

        public RequestBuilder url(String url) { // 设置URL
            this.url = url;
            return this; // 返回自身支持链式调用
        }

        public RequestBuilder method(String method) { // 设置方法
            this.method = method;
            return this;
        }

        public RequestBuilder timeout(Duration timeout) { // 设置超时
            this.timeout = timeout;
            return this;
        }

        public RequestBuilder header(String key, String value) { // 添加Header
            headers.put(key, value);
            return this;
        }

        public RequestBuilder param(String key, String value) { // 添加参数
            params.put(key, value);
            return this;
        }

        public HttpRequest build() { // 构建HttpRequest

            HttpRequest.Builder builder = HttpRequest.newBuilder(); // 创建Builder

            builder.uri(URI.create(buildUrl())); // 构建URI

            builder.timeout(timeout); // 设置超时

            headers.forEach(builder::header); // 设置Header

            if ("POST".equalsIgnoreCase(method)) { // POST请求

                StringJoiner sj = new StringJoiner("&"); // 拼接表单

                params.forEach((k, v) -> { // 遍历参数
                    sj.add(URLEncoder.encode(k, StandardCharsets.UTF_8) + "="
                            + URLEncoder.encode(v, StandardCharsets.UTF_8));
                });

                builder.POST(HttpRequest.BodyPublishers.ofString(sj.toString())); // 设置Body

            } else {
                builder.GET(); // GET请求
            }

            return builder.build(); // 返回请求对象
        }

        private String buildUrl() { // 构建URL

            if ("GET".equalsIgnoreCase(method) && !params.isEmpty()) {

                StringJoiner sj = new StringJoiner("&"); // 拼接参数

                params.forEach((k, v) -> {
                    sj.add(URLEncoder.encode(k, StandardCharsets.UTF_8) + "="
                            + URLEncoder.encode(v, StandardCharsets.UTF_8));
                });

                return url + "?" + sj; // 返回拼接URL
            }

            return url; // 无参数返回原URL
        }
    }

    /**
     * ===============================
     * 统一响应对象
     * ===============================
     */
    public static class HttpResult { // 定义响应模型

        private final int status; // 状态码
        private final String body; // 响应体
        private final Map<String, java.util.List<String>> headers; // 响应头

        public HttpResult(int status,
                          String body,
                          Map<String, java.util.List<String>> headers) {

            this.status = status; // 设置状态码
            this.body = body; // 设置响应体
            this.headers = headers; // 设置响应头
        }

        public int getStatus() { return status; } // 获取状态码

        public String getBody() { return body; } // 获取响应体

        public Map<String, java.util.List<String>> getHeaders() {
            return headers; // 获取响应头
        }

        public boolean is2xx() { // 判断是否成功
            return status >= 200 && status < 300;
        }
    }
}
