import os
import logging

from constants import *
from CommonUtils import CommonUtils
from ai_create_junit import *
import toYaml


def main():
    cur_path = os.getcwd()

    default_level = logging.INFO
    match LOG_LEVEL.lower():
        case "debug":
            default_level = logging.DEBUG
        case "warning":
            default_level = logging.WARNING
        case "error":
            default_level = logging.ERROR

    CommonUtils.setupLogger(os.path.join(cur_path, LOG_DIRECTORY), default_level)
    CommonUtils.writeLog(logging.INFO, "------------------------------")
    CommonUtils.writeLog(logging.INFO, "Junit生成処理を開始します。")
    CommonUtils.writeLog(logging.DEBUG, f"【ENDPOINT】 - {CHAT_DEPLOYMENT} に接続します。")

    try:
        in_path = os.path.join(cur_path, INPUT_DIRECTORY)
        pcl_files = CommonUtils.get_files_by_extension(in_path, ".xlsx")
        if not pcl_files:
            CommonUtils.writeLog(
                logging.ERROR,
                f"【{in_path}】に処理対象PCLがありません。処理を終了します。",
            )
            exit()

        total = len(pcl_files)
        count = 0
        for pcl_file in pcl_files:
            count += 1
            CommonUtils.writeLog(logging.INFO, f"PCL処理件数 {count}/{total} 処理中")

            folder_path = os.path.dirname(pcl_file)

            CommonUtils.writeLog(
                logging.DEBUG,
                f"フォルダ【{folder_path}】からソースファイルを読み込み、マージします。",
            )
            try:
                merged_sources = CommonUtils.merge_files(
                    folder_path, SRC_FILE_PATTERN, ENCODING_JAVA_SOURCE
                )
                CommonUtils.writeLog(
                    logging.DEBUG, f"merged_sources length: {len(merged_sources)}"
                )
            except Exception as e:
                CommonUtils.writeLog(
                    logging.WARNING,
                    f"ソースファイルのマージ処理中にエラーが発生しました：{e}",
                )
                continue

            if not merged_sources:
                CommonUtils.writeLog(
                    logging.WARNING, f"【{folder_path}】にソースファイルがありません。"
                )
                continue

            CommonUtils.writeLog(
                logging.DEBUG, f"【{pcl_file}】をYAML形式に変換します。"
            )
            pcl_yaml, class_name, method_name = toYaml.create_yaml(pcl_file)
            CommonUtils.writeLog(logging.DEBUG, f'pcl_file:\n{pcl_yaml}')

            method_name = f"{method_name[0].upper()}{method_name[1:]}"

            junit_file_name = f"{class_name}{method_name}Test.java"
            out_path = os.path.join(cur_path, OUTPUT_DIRECTORY)
            junit_file_path = os.path.join(out_path, junit_file_name)

            ai_create_junit(
                class_name, method_name, pcl_yaml, merged_sources, junit_file_path
            )

        CommonUtils.writeLog(logging.INFO, "Junit生成処理が終了しました。")

    except Exception as e:
        import traceback

        CommonUtils.writeLog(logging.ERROR, f"Junit生成処理中にエラーが発生しました：{e}")
        CommonUtils.writeLog(logging.ERROR, str(e))
        traceback.print_exc()


if __name__ == "__main__":
    main()
